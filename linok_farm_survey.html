// ================================================================ // LINOK — Pre-Installation Survey · Google Apps Script // הדבק את כל הקוד הזה ב-script.google.com // הוראות הגדרה בתחתית הקובץ // ================================================================ // ── הגדרות — מלא אחרי הרצת setup() ──────────────────────────── var SPREADSHEET_ID = ''; // ← הדבק כאן את ה-ID של ה-Sheet var PHOTOS_FOLDER_ID = ''; // ← הדבק כאן את ה-ID של תיקיית Drive // ─────────────────────────────────────────────────────────────── // ================================================================ // SETUP — הרץ פעם אחת כדי ליצור Sheet + תיקיית Drive // בחר "setup" מהתפריט → לחץ ▶ Run // ================================================================ function setup() { // יצירת Google Sheet var ss = SpreadsheetApp.create('LINOK — Farm Survey Responses'); var sheet = ss.getActiveSheet(); sheet.setName('Responses'); // כותרות עמודות var headers = [ 'Timestamp', 'Farm Name', 'Contact Name', 'Phone', 'Email', 'Address', 'Parlor Type', 'Stalls', 'Rows', 'Herd Size', 'Daily Milkings', 'Schedule', 'HMS', 'Internet', 'Router Location', 'Power Outlets', 'Outlet Count', 'PoE Location', 'Generator', 'Tech Name', 'Tech Phone', 'Dist Router→PoE', 'Dist PoE→Nearest', 'Dist PoE→Farthest', 'Dist Entry→PoE', 'Cables', 'Wall Material', 'Wall Space', 'Entry Points', 'Gate Width', 'Floor Type', 'RFID Poles', 'Notes', 'Total Photos', 'Photos Folder Link' ]; sheet.appendRow(headers); // עיצוב שורת כותרת var hRange = sheet.getRange(1, 1, 1, headers.length); hRange.setBackground('#1B6CA8') .setFontColor('#ffffff') .setFontWeight('bold') .setFontSize(11); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 180); // Timestamp sheet.setColumnWidth(2, 160); // Farm Name sheet.setColumnWidth(35, 300); // Photos Folder Link // יצירת תיקיית Drive var folder = DriveApp.createFolder('LINOK — Survey Photos'); // הדפסת ה-IDs ל-Logs Logger.log('=== SETUP COMPLETE ==='); Logger.log('SPREADSHEET_ID = ' + ss.getId()); Logger.log('PHOTOS_FOLDER_ID = ' + folder.getId()); Logger.log(''); Logger.log('העתק את שני ה-IDs והדבק אותם בחלק ה-CONFIG בראש הקובץ'); Logger.log('אחר כך: Deploy → New deployment'); } // ================================================================ // doPost — נקודת הכניסה לטופס // ================================================================ function doPost(e) { try { var payload = JSON.parse(e.postData.contents); if (payload.action === 'saveForm') { return saveFormData(payload.data); } if (payload.action === 'uploadPhoto') { return uploadPhoto(payload); } return jsonOk({ status: 'error', message: 'Unknown action' }); } catch (err) { return jsonOk({ status: 'error', message: err.toString() }); } } // ================================================================ // שמירת נתוני הטופס ל-Sheets // ================================================================ function saveFormData(d) { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); var sheet = ss.getSheetByName('Responses'); // יצירת תת-תיקייה לרפת הזו (FarmName — YYYY-MM-DD) var farmFolder = getOrCreateFarmFolder(d.farm_name, d.timestamp); var row = [ d.timestamp, d.farm_name, d.contact_name, d.phone, d.email, d.address, d.parlor_type, d.stalls, d.rows, d.herd_size, d.daily_milkings, d.schedule, d.hms, d.internet, d.router_location, d.power_outlets, d.outlet_count, d.poe_location, d.generator, d.tech_name, d.tech_phone, d.dist_router_poe, d.dist_poe_near, d.dist_poe_far, d.dist_entry_poe, d.cables, d.wall_material, d.wall_space, d.entry_points, d.gate_width, d.floor_type, d.rfid_poles, d.notes, d.total_photos, farmFolder.getUrl() // קישור ישיר לתיקיית התמונות של הרפת ]; sheet.appendRow(row); // צביעת שורה לסירוגין לקריאות var lastRow = sheet.getLastRow(); if (lastRow % 2 === 0) { sheet.getRange(lastRow, 1, 1, row.length).setBackground('#f0f7ff'); } return jsonOk({ status: 'ok', folderUrl: farmFolder.getUrl() }); } // ================================================================ // העלאת תמונה לתיקיית Drive // ================================================================ function uploadPhoto(p) { var farmFolder = getOrCreateFarmFolder(p.farmName, p.timestamp); // תת-תיקייה לפי קטגוריה (Router, MIKA_Installation וכו') var catFolder = getOrCreateSubFolder(farmFolder, p.category); var blob = Utilities.newBlob( Utilities.base64Decode(p.data), p.mimeType, p.filename ); catFolder.createFile(blob); return jsonOk({ status: 'ok' }); } // ================================================================ // Helpers // ================================================================ // מחזיר (או יוצר) תיקייה בפורמט "FarmName — YYYY-MM-DD" function getOrCreateFarmFolder(farmName, timestamp) { var root = DriveApp.getFolderById(PHOTOS_FOLDER_ID); var date = timestamp ? timestamp.substring(0, 10) : Utilities.formatDate(new Date(), 'UTC', 'yyyy-MM-dd'); var name = (farmName || 'Unknown Farm') + ' — ' + date; return getOrCreateSubFolder(root, name); } // מחזיר תת-תיקייה קיימת או יוצר חדשה function getOrCreateSubFolder(parent, name) { var iter = parent.getFoldersByName(name); return iter.hasNext() ? iter.next() : parent.createFolder(name); } // מחזיר JSON response (mode:no-cors מחייב 200 תמיד) function jsonOk(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ================================================================ // הוראות הגדרה מלאות // ================================================================ // // שלב 1 — צור Apps Script // ───────────────────────── // 1. לך ל- https://script.google.com → New project // 2. מחק את הקוד הקיים והדבק את כל הקובץ הזה // 3. שמור (Ctrl+S), תן שם לפרויקט: "LINOK Survey" // // שלב 2 — הרץ setup // ────────────────── // 4. בתפריט הפונקציות (ליד כפתור ▶) בחר "setup" // 5. לחץ ▶ Run // 6. אשר הרשאות (Google תבקש גישה ל-Sheets ו-Drive) // 7. פתח View → Logs — תראה שני IDs // 8. הדבק אותם ב-CONFIG בראש הקובץ: // var SPREADSHEET_ID = '1ABC...';
